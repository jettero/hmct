function rl2id() {
    id=$(perl -MNumber::RecordLocator -e 'print Number::RecordLocator->new->decode($ARGV[0])' $1)
    echo $id
}

function id2rl() {
    rl=$(perl -MNumber::RecordLocator -e 'print Number::RecordLocator->new->encode($ARGV[0])' $1)
    echo $rl
}
